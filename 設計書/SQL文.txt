        staff_id	VARCHAR(6),
        location_id	VARCHAR(4),
        agreement_id	VARCHAR(4),
        road_money	INT,
        start_time	TIME NOT NULL,
        end_time	TIME NOT NULL,
        work_time	TIME NOT NULL,
        overtime_start	TIME NOT NULL,
